\begin{frame}{Financial Compliance Multi-Agent Pipeline}
\textbf{Flow:} Transaction CSV $\rightarrow$ Coordinator $\rightarrow$ Band Chatroom $\rightarrow$ Policy $\rightarrow$ Risk $\rightarrow$ Legal $\rightarrow$ Decision

\vspace{0.6em}
\textbf{Agent roles}
\begin{itemize}
    \item \textbf{Policy Agent}: AML, KYC, sanctions screening
    \item \textbf{Risk Agent}: risk scoring and anomaly detection
    \item \textbf{Legal Agent}: BSA, OFAC, FATF compliance review
    \item \textbf{Decision Agent}: final verdict + audit trail
\end{itemize}

\vspace{0.4em}
\textbf{Decision outputs:} AUTO-APPROVE, ENHANCED REVIEW, ESCALATE TO HUMAN, DECLINE
\end{frame}
